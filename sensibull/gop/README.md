# To run this project with Docker 
## 1. Install docker if not there into system
## 2.Go to directory
cd /Go/src/sensibull/gop/sensibull

## 2. Run 
docker-compose up -d

## 1. Start redis server with default host and port

cd /Go/src/sensibull/gop/sensibull
## 3 Run
main.go file

cd /Go/src/sensibull/gop/sensibull/worker
